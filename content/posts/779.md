### 连接池

<img src="/imgs/30/15.png" />

**src/seq.js**

```js
const Sequelize = require("sequelize");

const conf = {
  host: "localhost",
  dialect: "mysql",
};

// 线上环境使用连接池
conf.pool = {
  max: 5, // 连接池中最大的连接数量
  min: 0,
  idle: 10000, // 如果一个连接池 10s 之内没有使用就会被释放
};

const seq = new Sequelize("koa2_weibo_db", "root", "Mysql_2018", conf);

module.exports = seq;
```
