### 将新建的三级分类组件变为全局组件

**src/components/index.ts**

```ts
import Category from "./Category/index.vue";
//...
const allGlobalComponent: any = {
  //...
  Category,
};
export default {
  install(app: any) {
    //...
    Object.keys(allGlobalComponent).forEach((key) => {
      app.component(key, allGlobalComponent[key]);
    });
  },
};
```
